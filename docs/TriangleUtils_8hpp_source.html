22 inline m3d::vec3 closest_point_on_triangle(const m3d::vec3 &P,
23 const m3d::vec3 &A,
24 const m3d::vec3 &B,
25 const m3d::vec3 &C)
26 {
27 const m3d::vec3 AB = B - A, AC = C - A, AP = P - A;
28 const m3d::scalar d1 = m3d::dot(AB, AP), d2 = m3d::dot(AC, AP);
29 if (d1 <= 0.0 && d2 <= 0.0)
30 return A; // vertex A
31
32 const m3d::vec3 BP = P - B;
33 const m3d::scalar d3 = m3d::dot(AB, BP), d4 = m3d::dot(AC, BP);
34 if (d3 >= 0.0 && d4 <= d3)
35 return B; // vertex B
36
37 const m3d::scalar vc = d1 * d4 - d3 * d2;
38 if (vc <= 0.0 && d1 >= 0.0 && d3 <= 0.0)
39 return A + AB * (d1 / (d1 - d3)); // edge AB
40
41 const m3d::vec3 CP = P - C;
42 const m3d::scalar d5 = m3d::dot(AB, CP), d6 = m3d::dot(AC, CP);
43 if (d6 >= 0.0 && d5 <= d6)
44 return C; // vertex C
45
46 const m3d::scalar vb = d5 * d2 - d1 * d6;
47 if (vb <= 0.0 && d2 >= 0.0 && d6 <= 0.0)
48 return A + AC * (d2 / (d2 - d6)); // edge AC
49
50 const m3d::scalar va = d3 * d6 - d5 * d4;
51 if (va <= 0.0 && (d4 - d3) >= 0.0 && (d5 - d6) >= 0.0)
52 {
53 const m3d::scalar w = (d4 - d3) / ((d4 - d3) + (d5 - d6));
54 return B + (C - B) * w; // edge BC
55 }
56
57 // Inside face
58 const m3d::scalar denom = 1.0 / (va + vb + vc);
59 const m3d::scalar v = vb * denom, w = vc * denom;
60 return A + AB * v + AC * w;
61 }

80 inline m3d::vec3 closest_point_on_segment(const m3d::vec3 &P,
81 const m3d::vec3 &Q,
82 const m3d::vec3 &X)
83 {
84 const m3d::vec3 d = Q - P;
85 const m3d::scalar len2 = m3d::length_sq(d);
86 if (len2 < m3d::EPSILON)
87 return P;
88 const m3d::scalar t = m3d::clamp(m3d::dot(X - P, d) / len2, m3d::scalar(0), m3d::scalar(1));
89 return P + d * t;
90 }

100 inline m3d::scalar sq_dist_segment_triangle(const m3d::vec3 &P, const m3d::vec3 &Q,
101 const m3d::vec3 &A, const m3d::vec3 &B,
102 const m3d::vec3 &C)
103 {
104 m3d::scalar best = sq_dist_point_triangle(P, A, B, C);
105 best = m3d::min(best, sq_dist_point_triangle(Q, A, B, C));
106
107 // Triangle edge AB vs segment
108 auto seg_seg_sq = [](const m3d::vec3 &p1, const m3d::vec3 &p2,
109 const m3d::vec3 &q1, const m3d::vec3 &q2) -> m3d::scalar
110 {
111 const m3d::vec3 d1 = p2 - p1, d2 = q2 - q1, r = p1 - q1;
112 const m3d::scalar a = m3d::length_sq(d1);
113 const m3d::scalar e = m3d::length_sq(d2);
114 const m3d::scalar f = m3d::dot(d2, r);
115 m3d::scalar s, t;
116 if (a < m3d::EPSILON && e < m3d::EPSILON)
117 {
118 s = t = 0.0;
119 }
120 else if (a < m3d::EPSILON)
121 {
122 s = 0.0;
123 t = m3d::clamp(f / e, m3d::scalar(0), m3d::scalar(1));
124 }
125 else
126 {
127 const m3d::scalar c = m3d::dot(d1, r);
128 if (e < m3d::EPSILON)
129 {
130 t = 0.0;
131 s = m3d::clamp(-c / a, m3d::scalar(0), m3d::scalar(1));
132 }
133 else
134 {
135 const m3d::scalar b = m3d::dot(d1, d2);
136 const m3d::scalar denom = a * e - b * b;
137 if (denom > m3d::EPSILON)
138 s = m3d::clamp((b * f - c * e) / denom, m3d::scalar(0), m3d::scalar(1));
139 else
140 s = 0.0;
141 t = (b * s + f) / e;
142 if (t < 0.0)
143 {
144 t = 0.0;
145 s = m3d::clamp(-c / a, m3d::scalar(0), m3d::scalar(1));
146 }
147 else if (t > 1.0)
148 {
149 t = 1.0;
150 s = m3d::clamp((b - c) / a, m3d::scalar(0), m3d::scalar(1));
151 }
152 }
153 }
154 return m3d::length_sq((p1 + d1 * s) - (q1 + d2 * t));
155 };
156
157 best = m3d::min(best, seg_seg_sq(P, Q, A, B));
158 best = m3d::min(best, seg_seg_sq(P, Q, B, C));
159 best = m3d::min(best, seg_seg_sq(P, Q, C, A));
160 return best;
161 }

174 inline bool sphere_vs_triangle(const m3d::vec3 &centre, m3d::scalar radius,
175 const m3d::vec3 &A, const m3d::vec3 &B, const m3d::vec3 &C,
176 const m3d::vec3 &face_normal, // precomputed, unit
177 ContactManifold &manifold)
178 {
179 const m3d::vec3 closest = closest_point_on_triangle(centre, A, B, C);
180 const m3d::vec3 delta = centre - closest;
181 const m3d::scalar dist2 = m3d::length_sq(delta);
182 if (dist2 > radius * radius)
183 return false;
184
185 const m3d::scalar dist = m3d::sqrt(dist2);
186 if (dist > m3d::EPSILON)
187 {
188 manifold.normal = delta / dist;
189 manifold.points[0].penetration_depth = radius - dist;
190 manifold.points[0].position = closest;
191 manifold.num_points = 1;
192 }
193 else
194 {
195 // Centre is on or inside the face — use face normal
196 const m3d::scalar side = m3d::dot(face_normal, centre - A);
197 manifold.normal = (side >= 0.0) ? face_normal : -face_normal;
198 manifold.points[0].penetration_depth = radius + m3d::abs(side);
199 manifold.points[0].position = centre - manifold.normal * radius;
200 manifold.num_points = 1;
201 }
202 return true;
203 }
