113 inline void remove_joint(JointCollection& jc, ConstraintCollection& cc, uint32_t id)
114 {
115 // 1. Read child info before the joint slot is freed.
116 const uint32_t idx = jc.index_of(id);
117 const u_short c_count = jc.constraint_count[idx];
118 const ConstraintBlock& cb = jc.constraints[idx];
119
120 // 2. Remove each child constraint via its stable ID.
121 for (u_short k = 0; k < c_count; ++k)
122 cc.remove(cb.ids[k]);
123
124 // 3. Remove the joint itself.
125 jc.remove(id);
126 }
