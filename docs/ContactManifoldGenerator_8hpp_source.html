59 m3d::vec3 *out)
60 {
61 int out_n = 0;
62 for (int i = 0; i < n; ++i)
63 {
64 const m3d::vec3 &A = poly[i];
65 const m3d::vec3 &B = poly[(i + 1) % n];
66
67 const m3d::scalar dA = m3d::dot(A - plane_p, plane_n);
68 const m3d::scalar dB = m3d::dot(B - plane_p, plane_n);
69
70 if (dA >= 0.0)
71 out[out_n++] = A;

73 if ((dA > 0.0) != (dB > 0.0))
74 {
75 const m3d::scalar t = dA / (dA - dB);
76 out[out_n++] = A + (B - A) * t;
