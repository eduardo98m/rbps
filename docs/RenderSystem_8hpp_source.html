135 // Crop to the saved plot screen-rect.
136 // ImPlot returns screen coords in ImGui space (points); on HiDPI
137 // screens these may need to be scaled by GetIO().DisplayFramebufferScale.
138 // We use floor/ceil to stay within valid pixel bounds.

140 const float scale = io.DisplayFramebufferScale.x; // usually 1 or 2
141
142 Rectangle crop;
143 crop.x = std::floor(req.pos.x * scale);
144 crop.y = std::floor(req.pos.y * scale);
145 crop.width = std::ceil (req.size.x * scale);
146 crop.height = std::ceil (req.size.y * scale);
147
148 // Clamp to image bounds
149 if (crop.x < 0) crop.x = 0;
150 if (crop.y < 0) crop.y = 0;
151 if (crop.x + crop.width > full.width) crop.width = full.width - crop.x;
152 if (crop.y + crop.height > full.height) crop.height = full.height - crop.y;
153
154 ImageCrop(&full, crop);
155
156 // Optional upscale (nearest-neighbour is fine for a plot)
157 if (req.upscale > 1)
