166 inline mat3 mat3_cast(const quat &q)
167 {
168 mat3 m;
169 scalar x2 = q.x + q.x, y2 = q.y + q.y, z2 = q.z + q.z;
170 scalar xx = q.x * x2, xy = q.x * y2, xz = q.x * z2;
171 scalar yy = q.y * y2, yz = q.y * z2, zz = q.z * z2;
172 scalar wx = q.w * x2, wy = q.w * y2, wz = q.w * z2;
173
174 m[0] = {1 - (yy + zz), xy + wz, xz - wy};
175 m[1] = {xy - wz, 1 - (xx + zz), yz + wx};
176 m[2] = {xz + wy, yz - wx, 1 - (xx + yy)};
177 return m;
178 }
