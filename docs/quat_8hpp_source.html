154 {

157 if (m3d::abs(1.0 - mag_sq) < 2.107342e-08)

214 inline vec3 to_rpy(const quat &q)
215 {
216 quat n = normalize(q);
217
218 scalar sinr_cosp = 2.0 * (n.w * n.x + n.y * n.z);
219 scalar cosr_cosp = 1.0 - 2.0 * (n.x * n.x + n.y * n.y);
220 scalar roll = m3d::atan2(sinr_cosp, cosr_cosp);
221
222 scalar sinp = 2.0 * (n.w * n.y - n.z * n.x);
223
224 scalar pitch;
225 if (m3d::abs(sinp) >= 1.0)
226 pitch = std::copysign(M_PI / 2.0, sinp);
227 else
228 pitch = std::asin(sinp);
229
230 scalar siny_cosp = 2.0 * (n.w * n.z + n.x * n.y);
231 scalar cosy_cosp = 1.0 - 2.0 * (n.y * n.y + n.z * n.z);
232 scalar yaw = m3d::atan2(siny_cosp, cosy_cosp);
233
234 return {roll, pitch, yaw};
