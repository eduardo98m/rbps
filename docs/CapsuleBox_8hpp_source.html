44 tf_box.rotate_vector(m3d::vec3(1, 0, 0)),
45 tf_box.rotate_vector(m3d::vec3(0, 1, 0)),
46 tf_box.rotate_vector(m3d::vec3(0, 0, 1))};

58 const m3d::vec3 &axis = ax[i];

61 m3d::scalar dist = m3d::abs(m3d::dot(diff, axis));

64 if (overlap < 0.0)

73 }

107 manifold.num_points = 0;
108

110 {
111 // ── Face Contact (Capsule Segment vs Box Face)
112 // Clip the capsule segment against the 4 boundary planes of the box face
113 m3d::scalar t0 = 0.0, t1 = 1.0;
114
115 // Helper for Liang-Barsky line clipping
116 auto clip_line = [](m3d::scalar p, m3d::scalar q, m3d::scalar &tMin, m3d::scalar &tMax) -> bool
117 {
118 if (m3d::abs(p) < m3d::EPSILON)
119 return q >= 0.0;
120 m3d::scalar t = q / p;
121 if (p > 0.0)
122 {
123 if (t < tMax)
124 tMax = t;
125 }
126 else

129 tMin = t;
130 }

145
146 // Clip against Axis B boundaries

148 if (!clip_line(pB, e[iB] - valB, t0, t1))

154 m3d::vec3 pt0 = p1 + d * t0;
155 m3d::scalar depth0 = capsule.radius + e[best_feat] - m3d::dot(pt0 - tf_box.pos, best_normal);
156 if (depth0 > 0.0)

158 manifold.points[manifold.num_points].position = pt0 - best_normal * capsule.radius;
159 manifold.points[manifold.num_points].penetration_depth = depth0;
160 manifold.num_points++;
161 }
162
163 // If the segment is long enough inside the box, add the second point
164 if (t1 - t0 > m3d::EPSILON)
165 {
166 m3d::vec3 pt1 = p1 + d * t1;
167 m3d::scalar depth1 = capsule.radius + e[best_feat] - m3d::dot(pt1 - tf_box.pos, best_normal);
168 if (depth1 > 0.0)
169 {
170 manifold.points[manifold.num_points].position = pt1 - best_normal * capsule.radius;
171 manifold.points[manifold.num_points].penetration_depth = depth1;
172 manifold.num_points++;
173 }
174 }
175 }
176 else
177 {
178 // ── Edge Contact (Capsule Segment vs Box Edge)
179 // Single point contact. Find the closest point on the segment to the box edge.
180 int edge_axis = best_feat - 3;
181 m3d::vec3 v = ax[edge_axis]; // Direction of the box edge
182
183 // Find a point lying on the exact edge we are hitting
184 m3d::vec3 support_box = tf_box.pos +
185 ax[0] * ((m3d::dot(ax[0], best_normal) >= 0.0) ? e[0] : -e[0]) +
186 ax[1] * ((m3d::dot(ax[1], best_normal) >= 0.0) ? e[1] : -e[1]) +
187 ax[2] * ((m3d::dot(ax[2], best_normal) >= 0.0) ? e[2] : -e[2]);
188
189 // Closest point calculation from segment `p1 + t*d` to line `support_box + s*v`
190 m3d::vec3 r = p1 - support_box;
191 m3d::scalar a = m3d::length_sq(d);
192 m3d::scalar b = m3d::dot(d, v);
193 m3d::scalar f = m3d::dot(v, r);
194 m3d::scalar c_dot = m3d::dot(d, r);
195
196 m3d::scalar denom = a - b * b;
197 m3d::scalar t = (denom > m3d::EPSILON) ? m3d::clamp((b * f - c_dot) / denom, m3d::scalar(0), m3d::scalar(1)) : 0.5;
198
199 m3d::vec3 pt = p1 + d * t;
200
201 manifold.num_points = 1;
202 manifold.points[0].position = pt - best_normal * capsule.radius;
203 manifold.points[0].penetration_depth = best_depth;
204 }
205
206 // Failsafe if clipping accidentally discarded valid points
207 if (manifold.num_points == 0)
208 {
209 manifold.num_points = 1;
210 manifold.points[0].position = c - best_normal * capsule.radius;
211 manifold.points[0].penetration_depth = best_depth;
212 }
213
214 return true;
215 }
