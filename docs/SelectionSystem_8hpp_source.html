34 static inline AABB3 snap_aabb(const ColliderSnap &c)
35 {
36 const float px = (float)c.world_pos.x;
37 const float py = (float)c.world_pos.y;
38 const float pz = (float)c.world_pos.z;
39
40 return std::visit([&](auto &&sh) -> AABB3
41 {
42 using T = std::decay_t<decltype(sh)>;
43 float r = 0.5f;
44
45 if constexpr (std::is_same_v<T, SpherSnap>)
46 r = (float)sh.radius;
47 else if constexpr (std::is_same_v<T, BoxSnap>)
48 r = std::sqrt((float)(sh.half_extents.x * sh.half_extents.x +
49 sh.half_extents.y * sh.half_extents.y +
50 sh.half_extents.z * sh.half_extents.z));
51 else if constexpr (std::is_same_v<T, CapsuleSnap>)
52 r = (float)sh.radius + (float)sh.half_height;
53 else if constexpr (std::is_same_v<T, CylinderSnap>)
54 r = (float)sh.base_radius + (float)sh.half_height;
55 else if constexpr (std::is_same_v<T, ConeSnap>)
56 r = std::max((float)sh.radius, (float)sh.height);
57 else if constexpr (std::is_same_v<T, EllipsoidSnap>)
58 r = (float)std::max({sh.semi_axes.x, sh.semi_axes.y, sh.semi_axes.z});
59 else if constexpr (std::is_same_v<T, HeightmapSnap>)
60 r = std::max(sh.cols * (float)sh.cell_size,
61 sh.rows * (float)sh.cell_size) * 0.5f;
62 else if constexpr (std::is_same_v<T, PlaneSnap>)
63 r = 50.0f;
64 else
65 r = 0.5f;
66
67 return { px - r, py - r, pz - r, px + r, py + r, pz + r };
68 }, c.shape);

71 // ── Ray-AABB intersection ─────────────────────────────────────────────────
72 static inline bool ray_aabb_hit(Ray ray, const AABB3 &box, float &t_out)
73 {
74 float tmin = -std::numeric_limits<float>::max();
75 float tmax = std::numeric_limits<float>::max();
76
77 const float origins[3] = { ray.position.x, ray.position.y, ray.position.z };
78 const float dirs[3] = { ray.direction.x, ray.direction.y, ray.direction.z };
79 const float bmin[3] = { box.min_x, box.min_y, box.min_z };
80 const float bmax[3] = { box.max_x, box.max_y, box.max_z };
81
82 for (int i = 0; i < 3; ++i)
83 {
84 if (std::fabs(dirs[i]) < 1e-8f)
85 {
86 if (origins[i] < bmin[i] || origins[i] > bmax[i]) return false;
87 }
88 else
89 {
90 float t1 = (bmin[i] - origins[i]) / dirs[i];
91 float t2 = (bmax[i] - origins[i]) / dirs[i];
92 if (t1 > t2) { float tmp = t1; t1 = t2; t2 = tmp; }
93 tmin = std::max(tmin, t1);
94 tmax = std::min(tmax, t2);
95 if (tmax < tmin) return false;
96 }
97 }
98
99 if (tmax < 0.0f) return false;
100 t_out = tmin > 0.0f ? tmin : tmax;
101 return true;
102 }
