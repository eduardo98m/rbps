156 void poll(rbps::World &world)
157 {
158 bool do_step_once = step_once_requested_.exchange(
159 false, std::memory_order_acq_rel);
160 if (do_step_once && paused_)
161 step_once_pending_ = true;
162
163 Command cmd;
164 while (transport.poll_command(cmd))
165 {
166 // Channel-level: do NOT forward to dispatch_command().
167 if (auto *t = std::get_if<CmdTrackQuantity>(&cmd))
168 { handle_track_command(*t); continue; }
169 if (std::get_if<CmdPause>(&cmd))
170 { paused_ = true; continue; }
171 if (std::get_if<CmdResume>(&cmd))
172 { paused_ = false; continue; }
173
174 // Everything else mutates World.
175 dispatch_command(world, cmd);
176 }
177 }

233 void sample_tracked(const FrameSnapshot &snap)
234 {
235 for (auto &series : tracked_series_)
236 {
237 float value = 0.0f;
238 switch (series.target)
239 {
240 case TrackTarget::BodyLinearSpeed:
241 for (const auto &b : snap.bodies)
242 if (b.id == series.id)
243 value = (float)m3d::length(b.linear_velocity);
244 break;
245 case TrackTarget::BodyAngularSpeed:
246 for (const auto &b : snap.bodies)
247 if (b.id == series.id)
248 value = (float)m3d::length(b.angular_velocity);
249 break;
250 case TrackTarget::BodyPositionX:
251 for (const auto &b : snap.bodies)
252 if (b.id == series.id) value = (float)b.position.x;
253 break;
254 case TrackTarget::BodyPositionY:
255 for (const auto &b : snap.bodies)
256 if (b.id == series.id) value = (float)b.position.y;
257 break;
258 case TrackTarget::BodyPositionZ:
259 for (const auto &b : snap.bodies)
260 if (b.id == series.id) value = (float)b.position.z;
261 break;
262 case TrackTarget::ContactNormalLambda:
263 for (const auto &ct : snap.contacts)
264 if (ct.collider_a == series.id || ct.collider_b == series.id)
265 value += (float)ct.normal_lambda;
266 break;
267 case TrackTarget::ContactPenetrationDepth:
268 for (const auto &ct : snap.contacts)
269 if (ct.collider_a == series.id || ct.collider_b == series.id)
270 value = std::max(value, (float)ct.penetration_depth);
271 break;
272 case TrackTarget::JointPosition:
273 for (const auto &j : snap.joints)
274 if (j.id == series.id)
275 value = (float)j.current_position;
276 break;
277 case TrackTarget::ConstraintLambda:
278 for (const auto &c : snap.constraints)
279 if (c.id == series.id)
280 value = (float)c.lambda;
281 break;
282 default: break;
283 }
284 // Store sim_time (seconds) as the x-axis value — physically
285 // meaningful and independent of frame rate / substep count.
286 series.push(snap.frame_index, snap.sim_time, value);
287 }
288 }

290 void handle_track_command(const CmdTrackQuantity &cmd)
291 {
292 if (!cmd.enabled)
293 {
294 auto it = std::remove_if(tracked_series_.begin(), tracked_series_.end(),
295 [&](const TrackedSeries &s)
296 { return s.target == cmd.target && s.id == cmd.id; });
297 tracked_series_.erase(it, tracked_series_.end());
298 return;
299 }
300 for (const auto &s : tracked_series_)
301 if (s.target == cmd.target && s.id == cmd.id) return;
302
303 TrackedSeries ts;
304 ts.target = cmd.target;
305 ts.id = cmd.id;
306 ts.samples.reserve(TRACK_HISTORY_DEPTH);
307 tracked_series_.push_back(std::move(ts));
308 }
