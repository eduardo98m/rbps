178 void remove(IdType id)
179 {
180 if (!_size)
181 return;
182
183 const IdType slot = slot_of(id);
184 if (slot >= _next_slot)
185 return;
186
187 const IdType idx = _data_index[slot];
188 if (idx >= _size)
189 return;
190
191 if constexpr (GenerationBits > 0)
192 if (gen_of(_ids[idx]) != gen_of(id))
193 return;
194
195 const IdType last_idx = _size - 1;
196 const IdType last_id = _ids[last_idx];
197 const IdType last_slot = slot_of(last_id);
198
199 self().swap_elements(idx, last_idx); // CRTP call into concrete struct
200 _ids[idx] = _ids[last_idx];
201
202 _data_index[last_slot] = idx;
203 _data_index[slot] = INVALID;
204
205 if constexpr (GenerationBits > 0)
206 ++_generation[slot];
207
208 _free_slots[_n_free++] = slot;
209 --_size;
210 }
