181 inline ID get_free_id(Core& c)
182 {
183 if (c.metadata.size() > c.n_items)
184 {
185 // A previously freed slot is available; bump its validity_id so
186 // any old handles to the same ID are invalidated immediately.
187 ++c.metadata[c.n_items].validity_id;
188 return c.metadata[c.n_items].rid;
189 }
190 // No free slot — allocate a brand-new one.
191 const ID new_id = static_cast<ID>(c.n_items);
192 c.metadata.push_back({new_id, 0});
193 c.indexes.push_back(new_id);
194 return new_id;
195 }
196
197 template<size_t Capacity>
198 inline ID get_free_id(CoreFixed<Capacity>& c)
199 {
200 assert(c.n_items < Capacity && "CoreFixed capacity exceeded");
201 if (c.n_slots > c.n_items)
202 {
203 ++c.metadata[c.n_items].validity_id;
204 return c.metadata[c.n_items].rid;
205 }
206 const ID new_id = static_cast<ID>(c.n_items);
207 c.indexes[new_id] = new_id;
208 c.metadata[new_id] = {new_id, 0};
209 ++c.n_slots;
210 return new_id;
211 }

214 template<typename TCore, typename TSwapFn>
215 inline void erase_impl(TCore& c, ID id, TSwapFn&& swap_fn)
216 {
217 assert(c.n_items > 0 && "erase called on empty collection");
218 const size_t data_idx = static_cast<size_t>(c.indexes[id]);
219 const size_t last_idx = c.n_items - 1;
220
221 // Invalidate the erased slot *before* the swap so that the
222 // incremented validity_id ends up in the free-slot position.
223 ++c.metadata[data_idx].validity_id;
224
225 if (data_idx != last_idx)
226 {
227 const ID last_rid = c.metadata[last_idx].rid;
228
229 // 1. User swaps their data arrays (swap-and-pop pattern).
230 swap_fn(data_idx, last_idx);
231
232 // 2. Update bookkeeping so the moved item is still reachable.
233 std::swap(c.metadata[data_idx], c.metadata[last_idx]);
234 std::swap(c.indexes[id], c.indexes[last_rid]);
235 }
236
237 --c.n_items;
238 // NOTE: metadata and indexes arrays keep their allocated size;
239 // the freed slot at position n_items is reused on next add().
240 // Caller must pop_back on every SoA data array.
241 }
242}
