118 inline FrameSnapshot build_snapshot(const rbps::World &world,
119 uint64_t frame_index,
120 double sim_time)
121 {
122 FrameSnapshot snap;
123 snap.frame_index = frame_index;
124 snap.sim_time = sim_time;
125 snap.timestep = world.timestep;
126 snap.substeps = world.substeps;
127 snap.broad_phase_pairs = static_cast<uint32_t>(
128 world.broad_phase_state.pairs.size());
129
130 const auto &bc = world.bodies;
131 const auto &cc = world.colliders;
132 const auto &cl = world.contacts;
133 const auto &jc = world.joints;
134 const auto &co = world.constraints;
135
136 // ── Bodies ───────────────────────────────────────────────────────────
137 snap.bodies.reserve(bc.count());
138 for (uint32_t i = 0; i < bc.count(); ++i)
139 {
140 BodySnap b;
141 b.id = bc._ids[i]; // _ids[data_index] = stable ID
142 b.is_static = (bc.type[i] == rbps::BodyType::STATIC);
143 b.position = bc.position[i];
144 b.orientation = bc.orientation[i];
145 b.linear_velocity = bc.linear_velocity[i];
146 b.angular_velocity = bc.angular_velocity[i];
147 b.force = bc.force[i];
148 b.torque = bc.torque[i];
149 b.mass = bc.mass[i];
150 b.inverse_mass = bc.inverse_mass[i];
151 snap.bodies.push_back(b);
152 }
153
154 // ── Colliders ────────────────────────────────────────────────────────
155 snap.colliders.reserve(cc.count());
156 for (uint32_t i = 0; i < cc.count(); ++i)
157 {
158 const uint32_t body_slot = bc.index_of(cc.body_id[i]);
159 const m3d::vec3 bpos = bc.position[body_slot];
160 const m3d::quat brot = bc.orientation[body_slot];
161 const m3d::tf wtf = rbps::collider_world_tf(cc, i, bpos, brot);
162
163 ColliderSnap c;
164 c.id = cc._ids[i];
165 c.body_id = cc.body_id[i];
166 c.is_static = cc.is_static[i];
167 c.world_pos = wtf.pos;
168 c.world_rot = wtf.rot;
169 c.restitution = cc.restitution[i];
170 c.static_friction = cc.static_friction[i];
171 c.dynamic_friction = cc.dynamic_friction[i];
172 c.shape = detail::to_shape_snap(cc.shape[i]);
173 snap.colliders.push_back(c);
174 }
175
176 // ── Contacts ─────────────────────────────────────────────────────────
177 snap.contacts.reserve(cl.n_contacts);
178 for (uint32_t i = 0; i < cl.n_contacts; ++i)
179 {
180 ContactSnap ct;
181 ct.body_a = bc._ids[cl.body_a[i]];
182 ct.body_b = bc._ids[cl.body_b[i]];
183 ct.collider_a = cc._ids[cl.collider_a[i]];
184 ct.collider_b = cc._ids[cl.collider_b[i]];
185 ct.point_on_a = cl.point_on_a[i];
186 ct.point_on_b = cl.point_on_b[i];
187 ct.normal = cl.normal[i];
188 ct.penetration_depth = cl.penetration_depth[i];
189 ct.normal_lambda = cl.normal_lambda[i];
190 ct.tangent_lambda = cl.tangent_lambda[i];
191 ct.normal_force = cl.normal_force[i];
192 ct.tangent_force = cl.tangent_force[i];
193 ct.relative_velocity = cl.relative_velocity[i];
194 ct.active = cl.collision[i];
195 snap.contacts.push_back(ct);
196 }
197
198 // ── Joints ───────────────────────────────────────────────────────────
199 snap.joints.reserve(jc.count());
200 for (uint32_t i = 0; i < jc.count(); ++i)
201 {
202 const uint32_t b1_slot = bc.index_of(jc.body_1[i]);
203 const uint32_t b2_slot = bc.index_of(jc.body_2[i]);
204
205 JointSnap j;
206 j.id = jc._ids[i];
207 j.body_a = jc.body_1[i];
208 j.body_b = jc.body_2[i];
209 j.main_axis = jc.main_axis[i];
210 j.current_position = jc.current_position[i];
211 j.lower_limit = jc.lower_limit[i];
212 j.upper_limit = jc.upper_limit[i];
213 j.limited = jc.limited[i];
214 j.damping = jc.damping[i];
215
216 j.anchor_a = bc.position[b1_slot] + m3d::rotate(bc.orientation[b1_slot], jc.r_1[i]);
217 j.anchor_b = bc.position[b2_slot] + m3d::rotate(bc.orientation[b2_slot], jc.r_2[i]);
218
219 switch (jc.type[i])
220 {
221 case rbps::JointType::PRISMATIC:
222 j.type = JointTypeSnap::Prismatic;
223 break;
224 case rbps::JointType::REVOLUTE:
225 j.type = JointTypeSnap::Revolute;
226 break;
227 case rbps::JointType::FIXED:
228 j.type = JointTypeSnap::Fixed;
229 break;
230 }
231 snap.joints.push_back(j);
232 }
233
234 // ── Constraints ──────────────────────────────────────────────────────
235 snap.constraints.reserve(co.count());
236 for (uint32_t i = 0; i < co.count(); ++i)
237 {
238 ConstraintSnap cs;
239 cs.id = co._ids[i];
240 cs.body_a = static_cast<uint32_t>(co.body_1[i]);
241 cs.body_b = static_cast<uint32_t>(co.body_2[i]);
242 cs.direction = co.direction[i];
243 cs.magnitude = co.magnitude[i];
244 cs.lambda = co.lambda[i];
245 cs.compliance = co.compliance[i];
246 cs.type = (co.type[i] == rbps::ConstraintType::POSITIONAL)
247 ? ConstraintTypeSnap::Positional
248 : ConstraintTypeSnap::Rotational;
249 snap.constraints.push_back(cs);
250 }
251
252 return snap;
253 }
