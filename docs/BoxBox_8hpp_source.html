29 inline m3d::scalar obb_radius(const Box &box, const m3d::tf &tf, const m3d::vec3 &axis)
30 {
31 const m3d::vec3 ax = tf.rotate_vector(m3d::vec3(1, 0, 0));
32 const m3d::vec3 ay = tf.rotate_vector(m3d::vec3(0, 1, 0));
33 const m3d::vec3 az = tf.rotate_vector(m3d::vec3(0, 0, 1));
34 return m3d::abs(m3d::dot(ax, axis)) * box.half_extents.x +
35 m3d::abs(m3d::dot(ay, axis)) * box.half_extents.y +
36 m3d::abs(m3d::dot(az, axis)) * box.half_extents.z;
37 }

39 inline bool test_axis(const m3d::vec3 &axis,
40 const Box &a, const m3d::tf &tf_a,
41 const Box &b, const m3d::tf &tf_b,
42 const m3d::vec3 &centre_diff,
43 m3d::scalar &best_depth,
44 m3d::vec3 &best_normal,
45 int feature_id,
46 int &best_feature,
47 m3d::scalar bias = 0.0)
48 {
49 const m3d::scalar len_sq = m3d::length_sq(axis);
50 if (len_sq < m3d::EPSILON * m3d::EPSILON)
51 return true;
52
53 const m3d::vec3 norm_axis = axis / m3d::sqrt(len_sq);
54 const m3d::scalar rA = obb_radius(a, tf_a, norm_axis);
55 const m3d::scalar rB = obb_radius(b, tf_b, norm_axis);
56 const m3d::scalar dist = m3d::abs(m3d::dot(centre_diff, norm_axis));
57 const m3d::scalar overlap = rA + rB - dist;
58
59 if (overlap < 0.0)
60 return false;
61
62 if ((overlap + bias) < best_depth)
63 {
64 best_depth = overlap + bias;
65 best_normal = (m3d::dot(centre_diff, norm_axis) >= 0.0) ? norm_axis : -norm_axis;
66 best_feature = feature_id;
67 }
68 return true;
69 }

71 inline void get_best_face(const Box &box, const m3d::tf &tf, const m3d::vec3 &dir,
72 m3d::vec3 out_corners[4], m3d::vec3 &out_normal)
73 {
74 const m3d::vec3 axes[3] = {
75 tf.rotate_vector(m3d::vec3(1, 0, 0)),
76 tf.rotate_vector(m3d::vec3(0, 1, 0)),
77 tf.rotate_vector(m3d::vec3(0, 0, 1))};
78 const m3d::scalar half[3] = {box.half_extents.x, box.half_extents.y, box.half_extents.z};
79
80 int best_axis = 0;
81 m3d::scalar best_dot = m3d::abs(m3d::dot(axes[0], dir));
82 for (int i = 1; i < 3; ++i)
83 {
84 m3d::scalar d = m3d::abs(m3d::dot(axes[i], dir));
85 if (d > best_dot)
86 {
87 best_dot = d;
88 best_axis = i;
89 }
90 }
91
92 const m3d::scalar sign = (m3d::dot(axes[best_axis], dir) >= 0.0) ? 1.0 : -1.0;
93 out_normal = axes[best_axis] * sign;
94 const m3d::vec3 face_centre = tf.pos + out_normal * half[best_axis];
95
96 const int u = (best_axis + 1) % 3;
97 const int v = (best_axis + 2) % 3;
98
99 out_corners[0] = face_centre + axes[u] * half[u] + axes[v] * half[v];
100 out_corners[1] = face_centre - axes[u] * half[u] + axes[v] * half[v];
101 out_corners[2] = face_centre - axes[u] * half[u] - axes[v] * half[v];
102 out_corners[3] = face_centre + axes[u] * half[u] - axes[v] * half[v];
103 }

105 inline int clip_polygon_by_plane(const m3d::vec3 *poly, int n,
106 const m3d::vec3 &plane_n, const m3d::vec3 &plane_p,
107 m3d::vec3 *out)
108 {
109 int out_n = 0;
110 for (int i = 0; i < n; ++i)
111 {
112 const m3d::vec3 &A = poly[i];
113 const m3d::vec3 &B = poly[(i + 1) % n];
114
115 const m3d::scalar dA = m3d::dot(A - plane_p, plane_n);
116 const m3d::scalar dB = m3d::dot(B - plane_p, plane_n);
117
118 if (dA >= 0.0)
119 out[out_n++] = A;
120 if ((dA > 0.0) != (dB > 0.0))
121 {
122 const m3d::scalar t = dA / (dA - dB);
123 out[out_n++] = A + (B - A) * t;
124 }
125 }
126 return out_n;
127 }

130 inline int reduce_to_4(const m3d::vec3 *pts, const m3d::scalar *depths, int n,
131 m3d::vec3 *out_pts, m3d::scalar *out_depths)
132 {
133 if (n <= 4)
134 {
135 for (int i = 0; i < n; ++i)
136 {
137 out_pts[i] = pts[i];
138 out_depths[i] = depths[i];
139 }
140 return n;
141 }
142
143 int i0 = 0;
144 for (int i = 1; i < n; ++i)
145 if (depths[i] > depths[i0])
146 i0 = i;
147
148 int i1 = (i0 == 0) ? 1 : 0;
149 m3d::scalar best = -1.0;
150 for (int i = 0; i < n; ++i)
151 {
152 if (i == i0)
153 continue;
154 m3d::scalar d = m3d::length_sq(pts[i] - pts[i0]);
155 if (d > best)
156 {
157 best = d;
158 i1 = i;
159 }
160 }
161
162 int i2 = -1;
163 best = -1.0;
164 const m3d::vec3 seg = pts[i1] - pts[i0];
165 for (int i = 0; i < n; ++i)
166 {
167 if (i == i0 || i == i1)
168 continue;
169 m3d::scalar d = m3d::length_sq(m3d::cross(pts[i] - pts[i0], seg));
170 if (d > best)
171 {
172 best = d;
173 i2 = i;
174 }
175 }
176 if (i2 < 0)
177 {
178 i2 = 0;
179 while (i2 == i0 || i2 == i1)
180 ++i2;
181 }
182
183 // To maximize area, find the point furthest on the OPPOSITE side of the diagonal as i2
184 int i3 = -1;
185 m3d::scalar min_dot = 1.0;
186 const m3d::vec3 tri_n = m3d::cross(pts[i2] - pts[i0], seg);
187 for (int i = 0; i < n; ++i)
188 {
189 if (i == i0 || i == i1 || i == i2)
190 continue;
191 m3d::scalar d = m3d::dot(m3d::cross(pts[i] - pts[i0], seg), tri_n);
192 if (d < min_dot)
193 {
194 min_dot = d;
195 i3 = i;
196 }
197 }
198
199 // If all points are on the same side, fallback to finding the point furthest from i2
200 if (i3 < 0 || min_dot >= 0.0)
201 {
202 best = -1.0;
203 for (int i = 0; i < n; ++i)
204 {
205 if (i == i0 || i == i1 || i == i2)
206 continue;
207 m3d::scalar d = m3d::length_sq(pts[i] - pts[i2]);
208 if (d > best)
209 {
210 best = d;
211 i3 = i;
212 }
213 }
214 }
215 if (i3 < 0)
216 {
217 i3 = 0;
218 while (i3 == i0 || i3 == i1 || i3 == i2)
219 ++i3;
220 }
221
222 out_pts[0] = pts[i0];
223 out_depths[0] = depths[i0];
224 out_pts[1] = pts[i1];
225 out_depths[1] = depths[i1];
226 out_pts[2] = pts[i2];
227 out_depths[2] = depths[i2];
228 out_pts[3] = pts[i3];
229 out_depths[3] = depths[i3];
230 return 4;
231 }

257 return false;

267 return false;

273 manifold.num_points = 0;
274
275 // Face-face contact
276 if (best_feature < 6)

278 bool ref_is_A = (best_feature < 3);
279 const Box &ref_box = ref_is_A ? a : b;
280 const m3d::tf &ref_tf = ref_is_A ? tf_a : tf_b;
281 const Box &inc_box = ref_is_A ? b : a;

289
290 m3d::vec3 buf0[16], buf1[16];
291 int cnt = 4;
292 for (int i = 0; i < 4; ++i)
293 buf0[i] = inc_corners[i];
294
295 m3d::vec3 ref_center = (ref_corners[0] + ref_corners[1] + ref_corners[2] + ref_corners[3]) * 0.25f;
296
297 for (int i = 0; i < 4; ++i)

299 const m3d::vec3 edge = ref_corners[(i + 1) % 4] - ref_corners[i];

303 side_n = -side_n;
304

306 if (cnt == 0)

310 }
311
312 if (cnt == 0)

314

328 }
329 }

332 // If clipping missed entirely (e.g., grazing corner float errors), gracefully skip the frame
333 // instead of placing an explosive fallback contact at the center of mass.
334 if (keep_n == 0)
335 return false;

349 {
350 // Edge-edge contact
351 const int ei = (best_feature - 6) / 3;
352 const int ej = (best_feature - 6) % 3;
353
354 const m3d::scalar extA[3] = {a.half_extents.x, a.half_extents.y, a.half_extents.z};
355 const m3d::scalar extB[3] = {b.half_extents.x, b.half_extents.y, b.half_extents.z};
356
357 const m3d::vec3 pa = tf_a.pos + tf_a.rotate_vector(m3d::vec3(
358 (m3d::dot(ax[0], best_normal) >= 0 ? 1.0 : -1.0) * (ei == 0 ? 0.0 : extA[0]),
359 (m3d::dot(ax[1], best_normal) >= 0 ? 1.0 : -1.0) * (ei == 1 ? 0.0 : extA[1]),
360 (m3d::dot(ax[2], best_normal) >= 0 ? 1.0 : -1.0) * (ei == 2 ? 0.0 : extA[2])));
361
362 const m3d::vec3 pb = tf_b.pos + tf_b.rotate_vector(m3d::vec3(
363 (m3d::dot(bx[0], -best_normal) >= 0 ? 1.0 : -1.0) * (ej == 0 ? 0.0 : extB[0]),
364 (m3d::dot(bx[1], -best_normal) >= 0 ? 1.0 : -1.0) * (ej == 1 ? 0.0 : extB[1]),
365 (m3d::dot(bx[2], -best_normal) >= 0 ? 1.0 : -1.0) * (ej == 2 ? 0.0 : extB[2])));
366
367 m3d::vec3 d1 = ax[ei], d2 = bx[ej];
368 m3d::vec3 r = pa - pb;
369
370 m3d::scalar b_dot = m3d::dot(d1, d2);
371 m3d::scalar c_dot = m3d::dot(d1, r);
372 m3d::scalar f_dot = m3d::dot(d2, r);
373 m3d::scalar denom = 1.0f - b_dot * b_dot;
374
375 m3d::scalar s = 0.0f, t = 0.0f;
376 if (denom > m3d::EPSILON)
377 s = (b_dot * f_dot - c_dot) / denom;
378
379 // --- FIX: Interdependent Edge Clamping ---
380 m3d::scalar extA_val = extA[ei];
381 m3d::scalar extB_val = extB[ej];
382
383 // If s is clamped, we must recalculate t, otherwise we get fake torque
384 if (s < -extA_val || s > extA_val)
385 s = std::clamp(s, -extA_val, extA_val);
386
387 t = s * b_dot + f_dot;
388 if (t < -extB_val || t > extB_val)
389 {
390 t = std::clamp(t, -extB_val, extB_val);
391 // If t is clamped, recalculate s again to lock in the true closest point
392 s = std::clamp(t * b_dot - c_dot, -extA_val, extA_val);
393 }
394
395 manifold.num_points = 1;
396 manifold.points[0].position = (pa + d1 * s + pb + d2 * t) * 0.5f;
397 manifold.points[0].penetration_depth = best_depth;
398 }
399
400 return true;
401 }
