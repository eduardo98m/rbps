28 // Transform a local-space mesh triangle to world space using tf.
29 inline void mesh_world_triangle(const MeshData &md, uint32_t f,
30 const m3d::tf &tf,
31 m3d::vec3 &A, m3d::vec3 &B, m3d::vec3 &C,
32 m3d::vec3 &world_normal)
33 {
34 A = tf.pos + tf.rotate_vector(md.vertices[md.indices[f * 3 + 0]]);
35 B = tf.pos + tf.rotate_vector(md.vertices[md.indices[f * 3 + 1]]);
36 C = tf.pos + tf.rotate_vector(md.vertices[md.indices[f * 3 + 2]]);
37 world_normal = tf.rotate_vector(md.face_normals[f]);
38 }

40 // Find the deepest penetrating contact across all triangles.
41 // `per_tri` is called with (A, B, C, world_normal, candidate_contact).
42 template <typename PerTriFn>
43 inline bool mesh_test_all(const Mesh &mesh, const m3d::tf &tf_mesh,
44 PerTriFn per_tri, ContactManifold &out)
45 {
46 if (!mesh.data)
47 return false;
48 const MeshData &md = *mesh.data;
49
50 bool hit = false;
51 m3d::scalar best_depth = 0.0;
52 ContactManifold candidate;
53
54 for (uint32_t f = 0; f < md.face_count; ++f)
55 {
56 m3d::vec3 A, B, C, wn;
57 mesh_world_triangle(md, f, tf_mesh, A, B, C, wn);
58
59 if (per_tri(A, B, C, wn, candidate))
60 {
61 const m3d::scalar d = candidate.points[0].penetration_depth;
62 if (d > best_depth)
63 {
64 best_depth = d;
65 out = candidate;
66 hit = true;
67 }
68 }
69 }
70 return hit;
71 }

112 // ── Box vs Mesh ───────────────────────────────────────────────────────────
113 // Uses the face-normal "push" approach: for each mesh triangle, check if
114 // the box has any vertex on the "inside" of the face, and vice-versa.
115 // This is a simplification of the full 15-axis SAT; it handles the common
116 // cases of thin walls robustly. For edge-edge contacts use the GJK fallback
117 // (the primary CollisionAlgorithm<Box,Mesh> below) if you need full accuracy.
118 template <>
119 struct CollisionAlgorithm<Box, Mesh>
120 {
121 static bool test(const Box &box, const m3d::tf &tf_box,
122 const Mesh &mesh, const m3d::tf &tf_mesh,
123 ContactManifold &out)
124 {
125 // Represent the box as a sphere (bounding) for fast rejection, then
126 // test each passing triangle with sphere_vs_triangle on the box centre.
127 // For a correct full SAT over box faces + mesh edges, integrate BoxBox logic.
128 // Here we delegate via the sphere of the box's bounding radius.
129 const m3d::scalar r = m3d::length(box.half_extents); // bounding sphere
130
131 return detail::mesh_test_all(mesh, tf_mesh, [&](const m3d::vec3 &A, const m3d::vec3 &B, const m3d::vec3 &C, const m3d::vec3 &wn, ContactManifold &c) -> bool
132 {
133 // Quick face-normal projection of box onto triangle plane
134 const m3d::scalar centre_dist = m3d::dot(tf_box.pos - A, wn);
135 // Box half-extent projected on world-normal
136 const m3d::vec3 local_n = tf_box.inverse_rotate_vector(wn);
137 const m3d::scalar box_r =
138 m3d::abs(local_n.x) * box.half_extents.x +
139 m3d::abs(local_n.y) * box.half_extents.y +
140 m3d::abs(local_n.z) * box.half_extents.z;
141
142 if (centre_dist > box_r) return false; // separated on face normal
143
144 // Project box centre onto triangle plane and test if nearby
145 const m3d::vec3 proj = tf_box.pos - wn * centre_dist;
146 const m3d::vec3 closest = tri::closest_point_on_triangle(proj, A, B, C);
147 const m3d::scalar edge_d = m3d::length(proj - closest);
148
149 if (edge_d > box_r) return false;
150
151 c.normal = wn;
152 c.num_points = 1;
153 c.points[0].penetration_depth = box_r - edge_d;
154 c.points[0].position = closest;
155
156 return c.points[0].penetration_depth > 0.0; }, out);
157 }
158 };

164 // ── Mesh vs Mesh ─────────────────────────────────────────────────────────
165 // Brute-force: treat each face of mesh A as a sphere of radius 0
166 // and test against mesh B. Very expensive for large meshes.
167 // In production, replace with BVH + GJK/SAT per overlapping triangle pair.
168 template <>
169 struct CollisionAlgorithm<Mesh, Mesh>
170 {
171 static bool test(const Mesh &a, const m3d::tf &tf_a,
172 const Mesh &b, const m3d::tf &tf_b,
173 ContactManifold &out)
174 {
175 if (!a.data || !b.data)
176 return false;
177
178 bool hit = false;
179 m3d::scalar best_depth = 0.0;
180 ContactManifold candidate;
181 const MeshData &md_a = *a.data;
182
183 for (uint32_t fa = 0; fa < md_a.face_count; ++fa)
184 {
185 m3d::vec3 A, B, C, wn_a;
186 detail::mesh_world_triangle(md_a, fa, tf_a, A, B, C, wn_a);
187
188 // Test each vertex of face fa against mesh B
189 const m3d::vec3 verts[3] = {A, B, C};
190 for (const auto &vtx : verts)
191 {
192 if (detail::mesh_test_all(b, tf_b, [&](const m3d::vec3 &bA, const m3d::vec3 &bB, const m3d::vec3 &bC, const m3d::vec3 &wn_b, ContactManifold &c)
193 { return tri::sphere_vs_triangle(vtx, 0.0, bA, bB, bC, wn_b, c); }, candidate))
194 {
195 if (candidate.points[0].penetration_depth > best_depth)
196 {
197 best_depth = candidate.points[0].penetration_depth;
198 out = candidate;
199 hit = true;
200 }
201 }
202 }
203 }
204 return hit;
205 }
