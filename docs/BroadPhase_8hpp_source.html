228 inline void broad_phase_move_swept(BroadPhaseState &bp,
229 BPHandle h,
230 const AABB &tight_aabb,
231 const m3d::vec3 &linear_velocity,
232 m3d::scalar dt)
233 {
234 const m3d::scalar margin = bp.config.fat_margin;
235 const m3d::vec3 dv = linear_velocity * dt;
236
237 const m3d::vec3 expand_min(
238 margin + (dv.x < 0 ? -dv.x : m3d::scalar(0)),
239 margin + (dv.y < 0 ? -dv.y : m3d::scalar(0)),
240 margin + (dv.z < 0 ? -dv.z : m3d::scalar(0)));
241
242 const m3d::vec3 expand_max(
243 margin + (dv.x > 0 ? dv.x : m3d::scalar(0)),
244 margin + (dv.y > 0 ? dv.y : m3d::scalar(0)),
245 margin + (dv.z > 0 ? dv.z : m3d::scalar(0)));
246
247 broad_phase_move(bp, h, AABB{tight_aabb.min - expand_min,
248 tight_aabb.max + expand_max});
249 }
