150 const uint32_t *face_indices = nullptr,
151 uint32_t face_count = 0)

154 hd->vertices = vertices;
155 hd->face_indices = face_indices;
156 hd->face_offsets = nullptr;
157 hd->face_normals = nullptr;
158 hd->vert_count = vert_count;
159 hd->face_count = (face_indices ? face_count : 0u);
160 hd->poly_indices = nullptr;
161 hd->poly_offsets = nullptr;
162 hd->poly_normals = nullptr;
163 hd->poly_count = 0;
164 hd->vertex_to_polys_offsets = nullptr;
165 hd->vertex_to_polys_indices = nullptr;
166 hd->vertex_to_neighbors_offsets = nullptr;
167 hd->vertex_to_neighbors_indices = nullptr;
168 hd->poly_to_polys_offsets = nullptr;
169 hd->poly_to_polys_indices = nullptr;
170
171 if (hd->face_count > 0)
172 {
173 auto *norms = new m3d::vec3[hd->face_count];
174 for (uint32_t f = 0; f < hd->face_count; ++f)
175 {
176 const m3d::vec3 &A = vertices[face_indices[f * 3 + 0]];
177 const m3d::vec3 &B = vertices[face_indices[f * 3 + 1]];
178 const m3d::vec3 &C = vertices[face_indices[f * 3 + 2]];
179 norms[f] = m3d::normalize(m3d::cross(B - A, C - A));

183 // ── Merge coplanar triangles into logical polygon faces ───────
184 // Two triangles belong to the same face when their outward
185 // normals point in the same direction within `kCoplanarDot`.
186 // For a convex polytope, parallel-normal triangles ARE coplanar
187 // (no two distinct planes share a normal direction on a convex
188 // hull), so the normal test is sufficient — no plane-distance
189 // check needed.
190 constexpr m3d::scalar kCoplanarDot = 0.99995; // ~0.57°
191 std::vector<int> parent(face_count);
192 for (uint32_t i = 0; i < face_count; ++i) parent[i] = static_cast<int>(i);
193 auto find = [&parent](int x) {
194 while (parent[x] != x) { parent[x] = parent[parent[x]]; x = parent[x]; }
195 return x;
196 };
197 for (uint32_t i = 0; i < face_count; ++i)
198 for (uint32_t j = i + 1; j < face_count; ++j)
199 if (m3d::dot(norms[i], norms[j]) > kCoplanarDot)
200 {
201 const int ri = find(static_cast<int>(i));
202 const int rj = find(static_cast<int>(j));
203 if (ri != rj) parent[ri] = rj;
204 }
205
206 // Collect triangles per group and average normals.
207 std::vector<std::vector<uint32_t>> groups;
208 std::vector<int> root_to_group(face_count, -1);
209 for (uint32_t i = 0; i < face_count; ++i)
210 {
211 const int r = find(static_cast<int>(i));
212 if (root_to_group[r] < 0)
213 {
214 root_to_group[r] = static_cast<int>(groups.size());
215 groups.emplace_back();
216 }
217 groups[root_to_group[r]].push_back(i);
218 }
219
220 // For each group: walk the boundary of the union by collecting
221 // edges that appear in only one triangle of the group, then
222 // chain them tip-to-tail into one CCW cycle.
223 std::vector<uint32_t> all_indices;
224 std::vector<uint32_t> offsets;
225 std::vector<m3d::vec3> poly_norms_vec;
226 offsets.push_back(0);
227
228 for (const auto &group : groups)
229 {
230 // Gather directed edges; index by source vertex for chaining.
231 struct DirEdge { uint32_t a, b; };
232 std::vector<DirEdge> tri_edges;
233 tri_edges.reserve(group.size() * 3);
234 for (const uint32_t f : group)
235 {
236 const uint32_t a = face_indices[f * 3 + 0];
237 const uint32_t b = face_indices[f * 3 + 1];
238 const uint32_t c = face_indices[f * 3 + 2];
239 tri_edges.push_back({a, b});
240 tri_edges.push_back({b, c});
241 tri_edges.push_back({c, a});
242 }
243
244 // An edge (a→b) is INTERIOR iff (b→a) appears elsewhere.
245 std::vector<DirEdge> boundary;
246 boundary.reserve(tri_edges.size());
247 for (size_t i = 0; i < tri_edges.size(); ++i)
248 {
249 bool reversed_present = false;
250 for (size_t j = 0; j < tri_edges.size(); ++j)
251 {
252 if (i == j) continue;
253 if (tri_edges[j].a == tri_edges[i].b &&
254 tri_edges[j].b == tri_edges[i].a)
255 { reversed_present = true; break; }
256 }
257 if (!reversed_present) boundary.push_back(tri_edges[i]);
258 }
259
260 if (boundary.empty()) continue; // pathological — drop polygon
261
262 // Chain edges: start with boundary[0], find next whose `a`
263 // matches current `b`, repeat until we close the loop.
264 std::vector<bool> used(boundary.size(), false);
265 std::vector<uint32_t> loop;
266 loop.reserve(boundary.size());
267
268 used[0] = true;
269 loop.push_back(boundary[0].a);
270 uint32_t current = boundary[0].b;
271 for (size_t step = 1; step < boundary.size(); ++step)
272 {
273 bool found = false;
274 for (size_t k = 0; k < boundary.size(); ++k)
275 {
276 if (used[k]) continue;
277 if (boundary[k].a == current)
278 {
279 used[k] = true;
280 loop.push_back(boundary[k].a);
281 current = boundary[k].b;
282 found = true;
283 break;
284 }
285 }
286 if (!found) break; // open chain — keep what we have
287 }
288
289 if (loop.size() < 3) continue;
290
291 // Average the group's triangle normals to reduce FP noise.
292 m3d::vec3 avg_n(0, 0, 0);
293 for (const uint32_t f : group) avg_n = avg_n + norms[f];
294 const m3d::scalar an_len = m3d::length(avg_n);
295 if (an_len < m3d::EPSILON) continue;
296 avg_n = avg_n / an_len;
297
298 for (uint32_t idx : loop) all_indices.push_back(idx);
299 offsets.push_back(static_cast<uint32_t>(all_indices.size()));
300 poly_norms_vec.push_back(avg_n);
301 }
302
303 const uint32_t pcount = static_cast<uint32_t>(poly_norms_vec.size());
304 if (pcount > 0)
305 {
306 auto *pi = new uint32_t[all_indices.size()];
307 auto *po = new uint32_t[pcount + 1];
308 auto *pn = new m3d::vec3[pcount];
309 for (size_t k = 0; k < all_indices.size(); ++k) pi[k] = all_indices[k];
310 for (uint32_t k = 0; k <= pcount; ++k) po[k] = offsets[k];
311 for (uint32_t k = 0; k < pcount; ++k) pn[k] = poly_norms_vec[k];
312 hd->poly_indices = pi;
313 hd->poly_offsets = po;
314 hd->poly_normals = pn;
315 hd->poly_count = pcount;
316
317 // ── Adjacency tables used by the manifold pipeline ────────
318 // O(P²) edge-pairing pass on poly count P (typically ≤ 32).
319 // Rebuild with hash maps if vert_count > ~256.
320 std::vector<std::vector<uint32_t>> v2p(vert_count);
321 std::vector<std::vector<uint32_t>> v2n(vert_count);
322 std::vector<std::vector<uint32_t>> p2p(pcount);
323
324 for (uint32_t p = 0; p < pcount; ++p)
325 {
326 const uint32_t s = po[p];
327 const uint32_t e = po[p + 1];
328 const uint32_t n = e - s;
329 for (uint32_t i = 0; i < n; ++i)
330 {
331 const uint32_t a = pi[s + i];
332 const uint32_t b = pi[s + (i + 1) % n];
333 auto &vp = v2p[a];
334 if (std::find(vp.begin(), vp.end(), p) == vp.end())
335 vp.push_back(p);
336 auto &va = v2n[a];
337 if (std::find(va.begin(), va.end(), b) == va.end())
338 va.push_back(b);
339 auto &vb = v2n[b];
340 if (std::find(vb.begin(), vb.end(), a) == vb.end())
341 vb.push_back(a);
342 }
343 }
344
345 // Two CCW polygons share an edge when one has (a,b) and the
346 // other has (b,a). Scan all polygon pairs.
347 for (uint32_t p = 0; p < pcount; ++p)
348 {
349 const uint32_t s_p = po[p];
350 const uint32_t n_p = po[p + 1] - s_p;
351 for (uint32_t q = 0; q < pcount; ++q)
352 {
353 if (q == p) continue;
354 const uint32_t s_q = po[q];
355 const uint32_t n_q = po[q + 1] - s_q;
356 bool shared = false;
357 for (uint32_t i = 0; i < n_p && !shared; ++i)
358 {
359 const uint32_t a = pi[s_p + i];
360 const uint32_t b = pi[s_p + (i + 1) % n_p];
361 for (uint32_t j = 0; j < n_q; ++j)
362 {
363 const uint32_t c = pi[s_q + j];
364 const uint32_t d = pi[s_q + (j + 1) % n_q];
365 if (a == d && b == c) { shared = true; break; }
366 }
367 }
368 if (shared) p2p[p].push_back(q);
369 }
370 }

372 auto flatten = [](const std::vector<std::vector<uint32_t>> &src,
373 uint32_t **out_offs, uint32_t **out_idx) {
374 const uint32_t n = static_cast<uint32_t>(src.size());
375 auto *offs = new uint32_t[n + 1];
376 offs[0] = 0;
377 uint32_t total = 0;
378 for (uint32_t i = 0; i < n; ++i)
379 {
380 total += static_cast<uint32_t>(src[i].size());
381 offs[i + 1] = total;
382 }
383 uint32_t *idx = (total > 0) ? new uint32_t[total] : nullptr;
384 uint32_t k = 0;
385 for (uint32_t i = 0; i < n; ++i)
386 for (uint32_t v : src[i])
387 idx[k++] = v;
388 *out_offs = offs;
389 *out_idx = idx;
390 };

457 inline m3d::vec3 support(const ConvexHull &h, const m3d::vec3 &dir)
458 {
459 if (!h.data || h.data->vert_count == 0)
460 return m3d::vec3{};
461 m3d::vec3 best = h.data->vertices[0];
462 m3d::scalar best_d = m3d::dot(best, dir);
463 for (uint32_t i = 1; i < h.data->vert_count; ++i)
464 {
465 const m3d::scalar d = m3d::dot(h.data->vertices[i], dir);
466 if (d > best_d)
467 {
468 best_d = d;
469 best = h.data->vertices[i];
470 }
471 }
472 return best;
473 }

487 return AABB{};

490 const m3d::vec3 half = (local.max - local.min) * 0.5;
491
492 const m3d::mat3 R = m3d::mat3_cast(tf.rot);
493 const m3d::vec3 world_centre = tf.pos + tf.rotate_vector(centre);
494 const m3d::vec3 extent(
495 m3d::abs(R[0][0]) * half.x + m3d::abs(R[1][0]) * half.y + m3d::abs(R[2][0]) * half.z,
496 m3d::abs(R[0][1]) * half.x + m3d::abs(R[1][1]) * half.y + m3d::abs(R[2][1]) * half.z,
497 m3d::abs(R[0][2]) * half.x + m3d::abs(R[1][2]) * half.y + m3d::abs(R[2][2]) * half.z);
498 return {world_centre - extent, world_centre + extent};

556 inline int face_corners(const ConvexHull &h, const m3d::tf &tf,
557 const m3d::vec3 &dir, m3d::vec3 *out, int capacity)
558 {
559 if (!h.data || h.data->vert_count == 0 || capacity <= 0)
560 return 0;
561
562 if (h.data->face_count == 0 || !h.data->face_indices || !h.data->face_normals)
563 {
564 const m3d::vec3 local_dir = tf.inverse_rotate_vector(dir);
565 const m3d::vec3 sup_world = tf.transform_point(support(h, local_dir));
566 return get_generic_face_corners(sup_world, dir, representative_radius(h), out, capacity);
567 }
568
569 const m3d::vec3 local_dir = tf.inverse_rotate_vector(dir);
570
571 // Prefer the merged polygon set when available — these are the
572 // actual logical faces and avoid triangulation seams becoming
573 // false silhouette edges in Sutherland-Hodgman clipping.
574 if (h.data->poly_count > 0 && h.data->poly_normals && h.data->poly_indices && h.data->poly_offsets)
575 {
576 uint32_t best = 0;
577 m3d::scalar best_d = m3d::dot(h.data->poly_normals[0], local_dir);
578 for (uint32_t p = 1; p < h.data->poly_count; ++p)
579 {
580 const m3d::scalar d = m3d::dot(h.data->poly_normals[p], local_dir);
581 if (d > best_d) { best_d = d; best = p; }
582 }
583 const uint32_t start = h.data->poly_offsets[best];
584 const uint32_t end = h.data->poly_offsets[best + 1];
585 const int n = static_cast<int>(end - start);
586 const int written = (n < capacity) ? n : capacity;
587 for (int i = 0; i < written; ++i)
588 out[i] = tf.transform_point(h.data->vertices[h.data->poly_indices[start + i]]);
589 return written;
590 }
591
592 // No merged polygons (degenerate hull or merge failed): fall
593 // back to a single best-aligned triangle.
594 if (capacity < 3) return 0;
595 uint32_t best = 0;
596 m3d::scalar best_d = m3d::dot(h.data->face_normals[0], local_dir);
597 for (uint32_t f = 1; f < h.data->face_count; ++f)
598 {
599 const m3d::scalar d = m3d::dot(h.data->face_normals[f], local_dir);
600 if (d > best_d) { best_d = d; best = f; }
601 }
602
603 out[0] = tf.transform_point(h.data->vertices[h.data->face_indices[best * 3 + 0]]);
604 out[1] = tf.transform_point(h.data->vertices[h.data->face_indices[best * 3 + 1]]);
605 out[2] = tf.transform_point(h.data->vertices[h.data->face_indices[best * 3 + 2]]);
606 return 3;
607 }

619 inline m3d::scalar compute_volume(const ConvexHull &h)
620 {
621 if (!h.data)
622 return 0.0;
623
624 if (h.data->face_count == 0 || !h.data->face_indices)
625 {
626 const m3d::vec3 ext = h.data->local_aabb.max - h.data->local_aabb.min;
627 return ext.x * ext.y * ext.z;
628 }
629
630 m3d::scalar v6 = 0.0;
631 for (uint32_t f = 0; f < h.data->face_count; ++f)
632 {
633 const m3d::vec3 &A = h.data->vertices[h.data->face_indices[f * 3 + 0]];
634 const m3d::vec3 &B = h.data->vertices[h.data->face_indices[f * 3 + 1]];
635 const m3d::vec3 &C = h.data->vertices[h.data->face_indices[f * 3 + 2]];
636 v6 += m3d::dot(A, m3d::cross(B, C));
637 }
638 return v6 / 6.0;
639 }

656 inline m3d::smat3 compute_inertia_tensor(const ConvexHull &h)
657 {
658 if (!h.data)
659 return m3d::smat3();
660
661 if (h.data->face_count == 0 || !h.data->face_indices)
662 {
663#ifndef NDEBUG
664 static bool warned = false;
665 if (!warned)
666 {
667 warned = true;
668 std::fprintf(stderr,
669 "[rbc] ConvexHull inertia: face data missing; "
670 "using AABB approximation.\n");
671 }
672#endif
673 const m3d::vec3 ext = h.data->local_aabb.max - h.data->local_aabb.min;
674 const m3d::scalar mass = ext.x * ext.y * ext.z;
675 const m3d::scalar x2 = ext.x * ext.x;
676 const m3d::scalar y2 = ext.y * ext.y;
677 const m3d::scalar z2 = ext.z * ext.z;
678 return m3d::smat3(
679 (1.0 / 12.0) * mass * (y2 + z2),
680 (1.0 / 12.0) * mass * (x2 + z2),
681 (1.0 / 12.0) * mass * (x2 + y2),
682 0.0, 0.0, 0.0);
683 }
684
685 m3d::scalar total_v6 = 0.0;
686 m3d::vec3 first_moment(0.0, 0.0, 0.0);
687 m3d::scalar Ixx_o = 0.0, Iyy_o = 0.0, Izz_o = 0.0;
688 m3d::scalar Ixy_o = 0.0, Ixz_o = 0.0, Iyz_o = 0.0;
689
690 for (uint32_t f = 0; f < h.data->face_count; ++f)
691 {
692 const m3d::vec3 &A = h.data->vertices[h.data->face_indices[f * 3 + 0]];
693 const m3d::vec3 &B = h.data->vertices[h.data->face_indices[f * 3 + 1]];
694 const m3d::vec3 &C = h.data->vertices[h.data->face_indices[f * 3 + 2]];
695
696 const m3d::scalar det = m3d::dot(A, m3d::cross(B, C));
697 const m3d::scalar dV = det / 6.0;
698 total_v6 += det;
699
700 first_moment = first_moment + (A + B + C) * (dV * 0.25);
701
702 auto sum_sq = [det](m3d::scalar a, m3d::scalar b, m3d::scalar c) {
703 return (det / 60.0) * (a * a + b * b + c * c + a * b + a * c + b * c);
704 };
705 auto sum_prod = [det](m3d::scalar ax, m3d::scalar ay,
706 m3d::scalar bx, m3d::scalar by,
707 m3d::scalar cx, m3d::scalar cy) {
708 return (det / 120.0) * (
709 2.0 * (ax * ay + bx * by + cx * cy)
710 + ax * by + ay * bx
711 + ax * cy + ay * cx
712 + bx * cy + by * cx);
713 };
714
715 const m3d::scalar mxx = sum_sq(A.x, B.x, C.x);
716 const m3d::scalar myy = sum_sq(A.y, B.y, C.y);
717 const m3d::scalar mzz = sum_sq(A.z, B.z, C.z);
718 const m3d::scalar mxy = sum_prod(A.x, A.y, B.x, B.y, C.x, C.y);
719 const m3d::scalar mxz = sum_prod(A.x, A.z, B.x, B.z, C.x, C.z);
720 const m3d::scalar myz = sum_prod(A.y, A.z, B.y, B.z, C.y, C.z);
721
722 Ixx_o += myy + mzz;
723 Iyy_o += mxx + mzz;
724 Izz_o += mxx + myy;
725 Ixy_o -= mxy;
726 Ixz_o -= mxz;
727 Iyz_o -= myz;
728 }

740 Iyy_o - mass * (cx * cx + cz * cz),
741 Izz_o - mass * (cx * cx + cy * cy),
742 Ixy_o + mass * cx * cy,
743 Ixz_o + mass * cx * cz,
