50 const m3d::scalar dist = m3d::length(local_diff);

52 // ── Case 1: sphere centre is OUTSIDE the box ──────────────────────
53 if (dist > m3d::EPSILON)
54 {

58 // local_diff / dist points B→A.
59 // Negate for the required A→B convention (sphere → box).
60 out.num_points = 1;
61 out.normal = -tf_box.rotate_vector(local_diff / dist);
62 out.points[0].penetration_depth = sphere.radius - dist;
63 out.points[0].position = tf_box.transform_point(closest); // box surface contact
64 return true;
65 }
66
67 // ── Case 2: sphere centre is INSIDE the box ───────────────────────
68 // Find the face whose penetration is smallest — that is the exit direction.
69 // Penetration along each face axis = half_extent − |local_coord|
70 const m3d::scalar px = box.half_extents.x - m3d::abs(local_centre.x);
71 const m3d::scalar py = box.half_extents.y - m3d::abs(local_centre.y);
72 const m3d::scalar pz = box.half_extents.z - m3d::abs(local_centre.z);
73

75 if (px <= py && px <= pz)

77 else if (py <= px && py <= pz)

81

83 out.num_points = 1;
